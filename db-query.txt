create database horsedoku;
use horsedoku;

create table races(
race_id int primary key auto_increment,
race_name varchar(80),
distance int,
ground varchar(4)
);

create table horses(
horse_id int primary key auto_increment,
horse_name varchar(80),
horse_birth date,
colors varchar(80),
G1_wins int,
img_path varchar(200)
);

create table horse_raced(
horse_id int,
foreign key (horse_id) references horses(horse_id),
race_id int,
foreign key (race_id) references races(race_id),
position int,
 PRIMARY KEY (horse_id, race_id)
);
